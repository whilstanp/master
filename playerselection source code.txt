    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        String p1 = jTextField1.getText();
        jLabel25.setText(p1);
        float hp = Float.parseFloat(jTextField2.getText());
        jLabel26.setText(jTextField3.getText());
        float atk = Float.parseFloat(jTextField3.getText());
        jLabel27.setText(jTextField4.getText());
        float def = Float.parseFloat(jTextField4.getText());
        jLabel28.setText(jTextField5.getText());
        float evade = Float.parseFloat(jTextField5.getText());
        jLabel29.setText(jTextField6.getText());
        float critical = Float.parseFloat(jTextField6.getText());
        String p2 = jTextField7.getText();
        jLabel30.setText(p2);
        float hp2 = Float.parseFloat(jTextField8.getText());
        jLabel31.setText(jTextField9.getText());
        float atk2 = Float.parseFloat(jTextField9.getText());
        jLabel32.setText(jTextField10.getText());
        float def2 = Float.parseFloat(jTextField10.getText());
        jLabel33.setText(jTextField11.getText());
        float evade2 = Float.parseFloat(jTextField11.getText());
        jLabel34.setText(jTextField12.getText());
        float critical2 = Float.parseFloat(jTextField12.getText());
        jLabel37.setText(jTextField2.getText());
        jLabel38.setText(jTextField8.getText());
        int turn = 1;
        while(hp!=0 || hp2!=0){
            float defL,atkP,defP,atkL,hit,hit2;
            double heal,heal2;
            if(turn%2==1){
                defL = def/100;
                if(Math.random()*100 < evade) defL = 1;
                atkP = atk2;
                if(Math.random()*100 < critical2){
                    atkP *= 2;
                    jTextArea1.append(p2+" gives critical hit -> "+atkP+"\n");}
                hit2 = (1-defL) * atkP;
                if(hit2<=0){
                    heal = (Math.random()*50);
                    jTextArea1.append(p1+" berhasil mengelak serangan\n");
                    jTextArea1.append(p1+" gives itself "+heal+" percentage of its health\n");
                    heal = (heal/100)*hp;
                    jTextArea1.append(p1+" gives itself "+heal+" hp\n");
                    hp += heal;}
                else if(hit2>=0){
                    jTextArea1.append(p1+" kena serangan sebanyak "+hit2+" damage\n");
                    hp -= hit2;}
            }
            else if(turn%2==0){
                defP = def2/100;
                if(Math.random()*100 < evade2) defP = 1;
                atkL = atk;
                if(Math.random()*100 < critical){
                    atkL *= 2;
                    jTextArea1.append(p1+" gives critical hit -> "+atkL+"\n");}
                hit = (1-defP) * atkL;
                if(hit<=0){
                    heal2 = (Math.random()*50);
                    jTextArea1.append(p2+" berhasil mengelak serangan\n");
                    jTextArea1.append(p1+" gives itself "+heal2+" percentage of its health\n");
                    heal2 = (heal2/100)*hp2;
                    jTextArea1.append(p1+" gives itself "+heal2+" hp\n");
                    hp2 += heal2;}
                else if(hit>=0){
                    jTextArea1.append(p2+" kena serangan sebanyak "+hit+" damage\n");
                    hp2 -= hit;}
            }
            jTextArea1.append(p1+" hp "+hp+"\n");
            jTextArea1.append(p2+" hp "+hp2+"\n");
            turn+=1;
            if(hp<=0 || hp2<=0) break;}
        if(hp2<=0){jLabel39.setText(p1+" wins!!");}
        else if(hp<=0){jLabel40.setText(p2+" wins!!");}
    }
